НАВИГАЦИЯ ПО ФС, ПРОСМОТР СОДЕРЖИМОГО фС
----------------------------------------
#### Вывести содержимое текущей директории:
	ls		# list;
		-a	# показать скрытые (системные) файлы и подкаталоги;

#### Сменить текущую директорию:
	cd <name_dir/>		# Change Directory;
		~			# псевдоним домашней директории текущего пользователя;
		..			# жесткая ссылка на родительскую (вышестоящую) директорию;

-------------------------------
РАБОТА С ФАЙЛАМИ И ДИРЕКТОРИЯМИ
-------------------------------
#### Создать диреткорию:
	mkdir <name_dir>		# Make Directory;

#### Создать пустой файл (обновить дату пмодификации файла):
	touch <name_file>

#### Скопировать файл (дублировать с новым именем):
	cp <souce> <dist>		# Copy;

#### Переместить (переименовать) файл:
mv <souce> <dist>		# Move;

#### Удалить файл или директорию:
	rm <file_or_empty-dir>		# (Remove) Удалить файл или пустую директорию;
		-r <not-an-empty-dir>	# Удалить непустую директорию;

------------------------------------
РАБОТА С ТЕКСТОМ (СОДЕРЖИМЫМ ФАЙЛОВ)
------------------------------------
#### Вывести текст:
	echo "<text>"			# Вывести текст на экран (на устройство вывода по умолчанию);
		 "<text>" > <name_file>	# Внести (с перезаписью) указанный текст в файл;

#### Вывести содержимое файла на экран (на устройство вывода по умолчанию);
	cat <name_file>

ТЕКСТОВЫЕ РЕДАКТОРЫ
-------------------
#### Консольный редактор текста Vim:
vim (file_name)		# Открыть vim (можно указать какой файл открыть, иначе создаст временный пустой файл);
	i				# Переход в режим ввода текстаl;
	ESC				# Выход в режим навигации по содержимому файла;
		:			# Режим сохранения содержимого (или выхода);
		w			# Записать изменения;
		q			# Выйти;
		!			# Выполнить запись и/или выход - принудительно;

#### Подключиться к сетевому ресурсу через telnet:
telnet <IP_or_dns-name> <port>
	### Например можно подключиться к сайту и загрузить код стартовой страницы ###
		telnet lib.ru 80
		GET / HTTP/1.1
		Host: lib.ru

#### Посмотреть сетевые подключения и сокеты (как в Windows, так и в Linux):
netstat
	# Печать сетевых подключений, таблиц маршрутизации, статистики интерфейсов,
		маскарадных подключений и членства в многоадресной рассылке;
		
	#### Windows ####
		help netstat		# Справка;
		-a					# кроме установленных соединений и прослушиваемые (listen) порты;
		
	#### Linux ####
		--help				# Краткая справка;
		-i					# Отобразить прослушиваемые подключения;


ss				# Утилита для исследования сокетов;
				!!!! в Linux ключи у netstat и ss - одинаковые!!!!
Для сетевых подключений в Linux с помощью утилиты ss можно использовать такие опции:

	-V		# Version показать версию утилиты;
	-n		# Numeric не определять имена служб;
	-r		# Resolve определять сетевые имена адресов с помощью DNS;
	-a		# All отобразить все сокеты (открытые соединения);
	-l		# Listening показать только прослушиваемые сокеты;
	-o		# Options показать информацию таймера;
	-e		# Extended выводить расширенную информацию о сокете;
	-p		# Processes, показать процессы, использующие сокет;
	-i		# Internal, посмотреть внутреннюю информацию TCP;
	-s		# Summary, статистика использования сокета;
	-D		# экспортировать текущее состояние TCP сокетов в файл;
	-F		# работать с информацией, взятой из файла;
	
	# Кроме того, можно вывести сокеты только нужного протокола:

	-4, --ipv4				# только сокеты протокола IP версии 4;
	-6 --ipv6				# только сокеты протокола IP версии 6;
	-0, --packet			# только PACKET сокеты;
	-t, --tcp				# TCP сокеты;
	-u, --udp				# UDP сокеты;
	-d, --dhcp				# DHCP сокеты;
	-r, --raw				# RAW сокеты;
	-x, --unix				# UNIX сокеты;
	-f	<имя_протокола>		# Универсальный фильтр по имени протокола;
